## RSI (Relative Strength Index)

### Basic RSI

```mql5
//+------------------------------------------------------------------+
//|                                            RSI_MQ5.mq5          |
//+------------------------------------------------------------------+
#property copyright "Your Name"
#property link      "https://yoursite.com"
#property version   "1.00"
#property indicator_separate_window
#property indicator_minimum 0
#property indicator_maximum 100

#property indicator_plots 1
#property indicator_type1 DRAW_LINE
#property indicator_color1 Purple
#property indicator_width1 2

input int RSIPeriod = 14;

double ExtRSIBuffer[];

int OnInit()
{
   SetIndexBuffer(0, ExtRSIBuffer, INDICATOR_CALCULATIONS);
   PlotIndexSetInteger(0, PLOT_DRAW_BEGIN, RSIPeriod);
   PlotIndexSetString(0, PLOT_LABEL, "RSI(" + IntegerToString(RSIPeriod) + ")");
   
   IndicatorSetString(INDICATOR_SHORTNAME, "RSI(" + IntegerToString(RSIPeriod) + ")");
   IndicatorSetInteger(INDICATOR_DIGITS, 2);
   
   return(INIT_SUCCEEDED);
}

int OnCalculate(const int rates_total,
              const int prev_calculated,
              const datetime &time[],
              const double &open[],
              const double &high[],
              const double &low[],
              const double &close[],
              const long &tick_volume[],
              const long &volume[],
              const int &spread[])
{
   if (rates_total < RSIPeriod)
      return(0);
   
   int i, limit;
   double avg_gain = 0, avg_loss = 0;
   double rs = 0;
   
   if (prev_calculated == 0)
   {
      for (i = 1; i <= RSIPeriod; i++)
      {
         double change = close[i] - close[i-1];
         if (change > 0)
            avg_gain += change;
         else
            avg_loss += (-change);
      }
      
      avg_gain /= RSIPeriod;
      avg_loss /= RSIPeriod;
      
      limit = RSIPeriod;
   }
   else
   {
      avg_gain = ExtRSIBuffer[prev_calculated-1];
      avg_loss = ExtRSIBuffer[prev_calculated-1 + RSIPeriod];
      
      double change = close[rates_total-1] - close[rates_total-2];
      double gain = (change > 0) ? change : 0;
      double loss = (change < 0) ? (-change) : 0;
      
      avg_gain = (avg_gain * (RSIPeriod - 1) + gain) / RSIPeriod;
      avg_loss = (avg_loss * (RSIPeriod - 1) + loss) / RSIPeriod;
      
      limit = prev_calculated - 1;
   }
   
   ExtRSIBuffer[rates_total-1 + RSIPeriod] = avg_gain;
   ExtRSIBuffer[rates_total-1 + RSIPeriod * 2] = avg_loss;
   
   if (avg_loss != 0)
      rs = avg_gain / avg_loss;
   else
      rs = 0;
   
   ExtRSIBuffer[rates_total-1] = 100.0 - (100.0 / (1.0 + rs));
   
   return(rates_total);
}
```

## MACD

### Basic MACD

```mql5
//+------------------------------------------------------------------+
//| MACD Indicator for MQL5                                         |
//+------------------------------------------------------------------+
#property copyright "Your Name"
#property link      "https://yoursite.com"
#property version   "1.00"
#property indicator_separate_window
#property indicator_plots 3
#property indicator_type1 DRAW_LINE
#property indicator_color1 Blue
#property indicator_width1 2
#property indicator_type2 DRAW_LINE
#property indicator_color2 Orange
#property indicator_width2 2
#property indicator_type3 DRAW_HISTOGRAM
#property indicator_color3 Green

input int FastEMA = 12;
input int SlowEMA = 26;
input int SignalLine = 9;

double ExtMacdBuffer[];
double ExtSignalBuffer[];
double ExtHistBuffer[];

int OnInit()
{
   SetIndexBuffer(0, ExtMacdBuffer, INDICATOR_CALCULATIONS);
   SetIndexBuffer(1, ExtSignalBuffer, INDICATOR_CALCULATIONS);
   SetIndexBuffer(2, ExtHistBuffer, INDICATOR_CALCULATIONS);
   
   PlotIndexSetInteger(0, PLOT_DRAW_BEGIN, SlowEMA);
   PlotIndexSetString(0, PLOT_LABEL, "MACD");
   PlotIndexSetInteger(1, PLOT_DRAW_BEGIN, SlowEMA + SignalLine);
   PlotIndexSetString(1, PLOT_LABEL, "Signal");
   PlotIndexSetInteger(2, PLOT_DRAW_BEGIN, SlowEMA + SignalLine);
   PlotIndexSetString(2, PLOT_LABEL, "Histogram");
   
   IndicatorSetString(INDICATOR_SHORTNAME, "MACD");
   IndicatorSetInteger(INDICATOR_DIGITS, 5);
   
   return(INIT_SUCCEEDED);
}

int OnCalculate(const int rates_total,
              const int prev_calculated,
              const datetime &time[],
              const double &open[],
              const double &high[],
              const double &low[],
              const double &close[],
              const long &tick_volume[],
              const long &volume[],
              const int &spread[])
{
   if (rates_total < SlowEMA)
      return(0);
   
   double emaFast, emaSlow;
   
   for (int i = 0; i < rates_total; i++)
   {
      emaFast = iEMA(NULL, 0, FastEMA, PRICE_CLOSE, MODE_EMA, i);
      emaSlow = iEMA(NULL, 0, SlowEMA, PRICE_CLOSE, MODE_EMA, i);
      
      ExtMacdBuffer[i] = emaFast - emaSlow;
   }
   
   // Signal line
   for (int i = 0; i < rates_total; i++)
   {
      if (i == 0)
         ExtSignalBuffer[i] = ExtMacdBuffer[i];
      else
         ExtSignalBuffer[i] = ExtSignalBuffer[i-1] + 
                          2.0 / (SignalLine + 1) * 
                          (ExtMacdBuffer[i] - ExtSignalBuffer[i-1]);
      
      ExtHistBuffer[i] = ExtMacdBuffer[i] - ExtSignalBuffer[i];
   }
   
   return(rates_total);
}
```

## Stochastic Oscillator

### Basic Stochastic

```mql5
//+------------------------------------------------------------------+
//| Stochastic Oscillator for MQL5                                    |
//+------------------------------------------------------------------+
#property copyright "Your Name"
#property link      "https://yoursite.com"
#property version   "1.00"
#property indicator_separate_window
#property indicator_minimum 0
#property indicator_maximum 100
#property indicator_plots 2
#property indicator_type1 DRAW_LINE
#property indicator_color1 Blue
#property indicator_width1 2
#property indicator_type2 DRAW_LINE
#property indicator_color2 Orange
#property indicator_width2 2

input int KPeriod = 14;
input int DPeriod = 3;
input int Slowing = 3;
input int Shift = 0;

double ExtKBuffer[];
double ExtDBuffer[];

int OnInit()
{
   SetIndexBuffer(0, ExtKBuffer, INDICATOR_CALCULATIONS);
   SetIndexBuffer(1, ExtDBuffer, INDICATOR_CALCULATIONS);
   
   PlotIndexSetInteger(0, PLOT_DRAW_BEGIN, KPeriod + Slowing);
   PlotIndexSetString(0, PLOT_LABEL, "%K");
   PlotIndexSetInteger(1, PLOT_DRAW_BEGIN, KPeriod + DPeriod + Slowing - 1);
   PlotIndexSetString(1, PLOT_LABEL, "%D");
   
   IndicatorSetString(INDICATOR_SHORTNAME, "Stochastic");
   IndicatorSetInteger(INDICATOR_DIGITS, 2);
   
   return(INIT_SUCCEEDED);
}

int OnCalculate(const int rates_total,
              const int prev_calculated,
              const datetime &time[],
              const double &open[],
              const double &high[],
              const double &low[],
              const double &close[],
              const long &tick_volume[],
              const long &volume[],
              const int &spread[])
{
   int i, limit;
   double lowest, highest, k_val;
   
   if (rates_total < KPeriod + Slowing)
      return(0);
   
   for (limit = prev_calculated; limit < rates_total; limit++)
   {
      lowest = low[limit];
      highest = high[limit];
      
      for (i = 0; i < KPeriod; i++)
      {
         if (low[limit-i] < lowest)
            lowest = low[limit-i];
         if (high[limit-i] > highest)
            highest = high[limit-i];
      }
      
      if (highest != lowest)
         k_val = (close[limit] - lowest) / (highest - lowest) * 100.0;
      else
         k_val = 50.0;
      
      if (Slowing > 1)
         ExtKBuffer[limit] = (ExtKBuffer[limit-1] * (Slowing - 1) + k_val) / Slowing;
      else
         ExtKBuffer[limit] = k_val;
      
      ExtDBuffer[limit] = ExtKBuffer[limit];
      for (i = 1; i < DPeriod && limit >= i; i++)
         ExtDBuffer[limit] = (ExtDBuffer[limit-1] * (i) + ExtKBuffer[limit]) / (i + 1);
   }
   
   return(rates_total);
}
```

## CCI (Commodity Channel Index)

### CCI Indicator

```mql5
//+------------------------------------------------------------------+
//| CCI for MQL5                                                   |
//+------------------------------------------------------------------+
#property copyright "Your Name"
#property link      "https://yoursite.com"
#property version   "1.00"
#property indicator_separate_window

#property indicator_plots 1
#property indicator_type1 DRAW_LINE
#property indicator_color1 Blue
#property indicator_width1 2

input int CCIPeriod = 20;

double ExtCCIBuffer[];

int OnInit()
{
   SetIndexBuffer(0, ExtCCIBuffer, INDICATOR_CALCULATIONS);
   PlotIndexSetInteger(0, PLOT_DRAW_BEGIN, CCIPeriod);
   PlotIndexSetString(0, PLOT_LABEL, "CCI(" + IntegerToString(CCIPeriod) + ")");
   
   IndicatorSetString(INDICATOR_SHORTNAME, "CCI(" + IntegerToString(CCIPeriod) + ")");
   IndicatorSetInteger(INDICATOR_DIGITS, 2);
   
   return(INIT_SUCCEEDED);
}

int OnCalculate(const int rates_total,
              const int prev_calculated,
              const datetime &time[],
              const double &open[],
              const double &high[],
              const double &low[],
              const double &close[],
              const long &tick_volume[],
              const long &volume[],
              const int &spread[])
{
   if (rates_total < CCIPeriod)
      return(0);
   
   double tp[], sma, mean_dev;
   ArrayResize(tp, rates_total);
   
   // Calculate Typical Price
   for (int i = 0; i < rates_total; i++)
      tp[i] = (high[i] + low[i] + close[i]) / 3.0;
   
   for (int i = CCIPeriod - 1; i < rates_total; i++)
   {
      sma = 0;
      for (int j = 0; j < CCIPeriod; j++)
         sma += tp[i-j];
      sma /= CCIPeriod;
      
      mean_dev = 0;
      for (int j = 0; j < CCIPeriod; j++)
         mean_dev += MathAbs(tp[i-j] - sma);
      mean_dev /= CCIPeriod;
      
      if (mean_dev != 0)
         ExtCCIBuffer[i] = (tp[i] - sma) / (0.015 * mean_dev);
      else
         ExtCCIBuffer[i] = 0;
   }
   
   return(rates_total);
}
```

```mql5

input int RSIPeriod = 14;
input int StochK = 14;
input int StochD = 3;
input int CCIPeriod = 20;

double RSIBuffer[];
double StochKBuffer[];
double StochDBuffer[];
double CCIBuffer[];

int OnInit()
{
   SetIndexBuffer(0, RSIBuffer, INDICATOR_CALCULATIONS);
   SetIndexBuffer(1, StochKBuffer, INDICATOR_CALCULATIONS);
   SetIndexBuffer(2, StochDBuffer, INDICATOR_CALCULATIONS);
   SetIndexBuffer(3, CCIBuffer, INDICATOR_CALCULATIONS);
   
   PlotIndexSetString(0, PLOT_LABEL, "RSI");
   PlotIndexSetString(1, PLOT_LABEL, "Stoch %K");
   PlotIndexSetString(2, PLOT_LABEL, "Stoch %D");
   PlotIndexSetString(3, PLOT_LABEL, "CCI");
   
   IndicatorSetString(INDICATOR_SHORTNAME, "Oscillators");
   
   return(INIT_SUCCEEDED);
}

int OnCalculate(const int rates_total,
              const int prev_calculated,
              const datetime &time[],
              const double &open[],
              const double &high[],
              const double &low[],
              const double &close[],
              const long &tick_volume[],
              const long &volume[],
              const int &spread[])
{
   // RSI Calculation
   double avg_gain = 0, avg_loss = 0;
   for (int i = rates_total - RSIPeriod; i < rates_total; i++)
   {
      double change = close[i] - close[i-1];
      if (change > 0) avg_gain += change;
      else avg_loss += (-change);
   }
   avg_gain /= RSIPeriod;
   avg_loss /= RSIPeriod;
   
   double rs = (avg_loss != 0) ? avg_gain / avg_loss : 0;
   RSIBuffer[rates_total-1] = 100.0 - (100.0 / (1.0 + rs));
   
   // Stochastic Calculation
   double lowest = low[rates_total-1];
   double highest = high[rates_total-1];
   for (int j = 0; j < StochK; j++)
   {
      if (low[rates_total-1-j] < lowest) lowest = low[rates_total-1-j];
      if (high[rates_total-1-j] > highest) highest = high[rates_total-1-j];
   }
   double k = (highest != lowest) ? 
             ((close[rates_total-1] - lowest) / (highest - lowest)) * 100 : 50;
   StochKBuffer[rates_total-1] = k;
   StochDBuffer[rates_total-1] = k; // Simplified
   
   // CCI Calculation
   double tp = (high[rates_total-1] + low[rates_total-1] + close[rates_total-1]) / 3.0;
   double sma = 0;
   for (int i = 0; i < CCIPeriod; i++)
      sma += (high[rates_total-1-i] + low[rates_total-1-i] + close[rates_total-1-i]) / 3.0;
   sma /= CCIPeriod;
   
   double sum_dev = 0;
   for (int i = 0; i < CCIPeriod; i++)
      sum_dev += MathAbs(tp - ((high[rates_total-1-i] + low[rates_total-1-i] + close[rates_total-1-i]) / 3.0));
   double mean_dev = sum_dev / CCIPeriod;
   
   CCIBuffer[rates_total-1] = (mean_dev != 0) ? (tp - sma) / (0.015 * mean_dev) : 0;
   
   return(rates_total);
}
```

```mql5
   if (!IsNewBar())
      return;
```

## Complete EA with Multiple Oscillators

```mql5
//+------------------------------------------------------------------+
//| Multi-Oscillator EA                                             |
//+------------------------------------------------------------------+
#property copyright "Your Name"
#property link      "https://yoursite.com"
#property version   "1.00"
#property strict

input double Lots = 0.1;
input int RSIPeriod = 14;
input int FastEMA = 12;
input int SlowEMA = 26;
input int SignalPeriod = 9;
input int StochK = 14;
input int CCIPeriod = 20;
input ulong EA_Magic = 12345000;

double ExtRSIBuffer[];
double ExtMACDBuffer[];
double ExtSignalBuffer[];
double ExtHistBuffer[];
double ExtStochKBuffer[];
double ExtStochDBuffer[];

int OnInit()
{
   SetIndexBuffer(0, ExtRSIBuffer, INDICATOR_CALCULATIONS);
   SetIndexBuffer(1, ExtMACDBuffer, INDICATOR_CALCULATIONS);
   SetIndexBuffer(2, ExtSignalBuffer, INDICATOR_CALCULATIONS);
   SetIndexBuffer(3, ExtHistBuffer, INDICATOR_CALCULATIONS);
   SetIndexBuffer(4, ExtStochKBuffer, INDICATOR_CALCULATIONS);
   SetIndexBuffer(5, ExtStochDBuffer, INDICATOR_CALCULATIONS);
   
   Print("Multi-Oscillator EA initialized");
   return(INIT_SUCCEEDED);
}

void OnTick()
{
   if (IsNewBar())
      ProcessSignals();
}

bool IsNewBar()
{
   static datetime lastBar = 0;
   datetime currentBar = iTime(Symbol(), PERIOD_CURRENT, 0);
   
   if (lastBar != currentBar)
   {
      lastBar = currentBar;
      return true;
   }
   
   return false;
}

void ProcessSignals()
{
   // RSI
   double rsi = iRSI(NULL, 0, RSIPeriod, PRICE_CLOSE, 0);
   double prev_rsi = iRSI(NULL, 0, RSIPeriod, PRICE_CLOSE, 1);
   
   // MACD
   double macd = iMACD(NULL, 0, FastEMA, SlowEMA, SignalPeriod, PRICE_CLOSE, 0);
   double signal = iMACD(NULL, 0, FastEMA, SlowEMA, SignalPeriod, PRICE_CLOSE, 1);
   
   // Stochastic
   double stochK = iStochastic(NULL, 0, StochK, 3, 3, MODE_SMA, 0, 0);
   double stochD = iStochastic(NULL, 0, StochK, 3, 3, MODE_SMA, 0, 1);
   
   // CCI
   double cci = iCCI(NULL, 0, CCIPeriod, PRICE_CLOSE, 0);
   
   // Buy conditions: RSI > 30, MACD > signal, Stochastic < 20
   bool buyCondition = (rsi > 30 && prev_rsi <= 30) || 
                    (macd > signal) || 
                    (stochK < 20);
   
   // Sell conditions: RSI < 70, MACD < signal, Stochastic > 80
   bool sellCondition = (rsi < 70 && prev_rsi >= 70) || 
                     (macd < signal) || 
                     (stochK > 80);
   
   if (buyCondition)
      ExecuteTrade(ORDER_TYPE_BUY);
   else if (sellCondition)
      ExecuteTrade(ORDER_TYPE_SELL);
}

void ExecuteTrade(ENUM_ORDER_TYPE orderType)
{
   MqlTradeRequest request = {};
   MqlTradeResult result = {};
   
   request.action = TRADE_ACTION_DEAL;
   request.symbol = Symbol();
   request.volume = Lots;
   request.type = orderType;
   request.price = (orderType == ORDER_TYPE_BUY) ? 
                SymbolInfoDouble(Symbol(), SYMBOL_ASK) : 
                SymbolInfoDouble(Symbol(), SYMBOL_BID);
   request.deviation = 10;
   request.magic = EA_Magic;
   request.comment = "Multi-Oscillator EA";
   
   bool sent = OrderSend(request, result);
   
   if (sent && result.retcode == TRADE_RETCODE_DONE)
   {
      Print("Order executed: ", orderType == ORDER_TYPE_BUY ? "BUY" : "SELL");
   }
   else
   {
      Print("Order failed: ", result.retcode);
   }
}
```

MQL5 provides improved built-in functions compared to MQL4:
- RSI: `iRSI()`
- MACD: `iMACD()` (returns close, signal, histogram)
- Stochastic: `iStochastic()` (returns %K, %D)
- CCI: `iCCI()`
- Williams %R: Manual calculation needed

Key improvements in MQL5:
- Object-oriented buffer handling
- Better array management
- Enhanced trade functions with MqlTradeRequest
- Better error handling
- Multiple timeframe support

Use built-in indicators for reliability, and custom indicators only when specific calculations are needed. This article provides the foundation for building oscillator-based strategies in MetaTrader 5.
