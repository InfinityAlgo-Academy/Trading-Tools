## RSI (Relative Strength Index)

### Basic RSI

```mql4
//+------------------------------------------------------------------+
//|                                                              |
//+------------------------------------------------------------------+
#property copyright "Your Name"
#property link      "https://yoursite.com"
#property version   "1.00"
#property strict

#property indicator_separate_window
#property indicator_minimum 0
#property indicator_maximum 100

input int RSIPeriod = 14;
input double Overbought = 70.0;
input double Oversold = 30.0;

double ExtRSIBuffer[];
double ExtCloseBuffer[];

int OnInit()
{
   string short_name = "RSI(" + IntegerToString(RSIPeriod) + ")";
   IndicatorShortName(short_name);
   
   IndicatorSetInteger(INDICATOR_DIGITS, 2);
   IndicatorSetString(INDICATOR_SHORTNAME, short_name);
   
   SetIndexBuffer(0, ExtRSIBuffer, INDICATOR_CALCULATIONS);
   SetIndexStyle(0, DRAW_LINE);
   SetIndexLabel(0, "RSI");
   
   ArraySetAsSeries(ExtRSIBuffer, true);
   ArraySetAsSeries(ExtCloseBuffer, true);
   
   return(INIT_SUCCEEDED);
}

int OnCalculate(const int rates_total,
                const int prev_calculated,
                const datetime &time[],
                const double &open[],
                const high[],
                const low[],
                const close[],
                const long &volume[],
                const int &spread[])
{
   if (rates_total < RSIPeriod)
      return(0);
   
   ArraySetAsSeries(close, true);
   ArraySetAsSeries(high, true);
   ArraySetAsSeries(low, true);
   
   int i, limit;
   double avg_gain = 0, avg_loss = 0;
   double gain, loss;
   
   if (prev_calculated == 0)
   {
      for (i = 0; i < RSIPeriod; i++)
         ExtRSIBuffer[i] = 0.0;
      
      avg_gain = 0; avg_loss = 0;
      for (i = 1; i <= RSIPeriod; i++)
      {
         gain = close[i] - close[i-1];
         if (gain < 0) loss = -gain; else loss = 0;
         avg_gain += gain;
         avg_loss += loss;
      }
      avg_gain /= RSIPeriod;
      avg_loss /= RSIPeriod;
      
      limit = RSIPeriod;
   }
   else
   {
      avg_gain = ExtRSIBuffer[prev_calculated-1];
      avg_loss = ExtRSIBuffer[prev_calculated-1 + RSIPeriod];
      
      limit = prev_calculated - 1;
   }
   
   for (i = limit; i < rates_total; i++)
   {
      gain = close[i] - close[i-1];
      loss = 0;
      if (gain < 0)
      {
         loss = -gain;
         gain = 0;
      }
      
      avg_gain = (avg_gain * (RSIPeriod - 1) + gain) / RSIPeriod;
      avg_loss = (avg_loss * (RSIPeriod - 1) + loss) / RSIPeriod;
      
      if (avg_loss != 0)
         ExtRSIBuffer[i] = 100.0 - (100.0 / (1.0 + avg_gain / avg_loss));
      else
         ExtRSIBuffer[i] = 100.0;
   }
   
   return(rates_total);
}
```

```mql4
                const close[],

   {
      for (int i = 0; i < RSIPeriod; i++)
      {
         double change = close[i] - close[i-1];
         if (change > 0) avg_gain += change;
         else avg_loss += (-change);
      }
      avg_gain /= RSIPeriod;
      avg_loss /= RSIPeriod;
   }
   else
   {
      avg_gain = ExtRSIBuffer[prev_calculated-1];
      avg_loss = ExtRSIBuffer[prev_calculated-1 + RSIPeriod];
      
      double change = close[rates_total-1] - close[rates_total-2];
      double gain = (change > 0) ? change : 0;
      double loss = (change < 0) ? (-change) : 0;
      
      avg_gain = (avg_gain * (RSIPeriod-1) + gain) / RSIPeriod;
      avg_loss = (avg_loss * (RSIPeriod-1) + loss) / RSIPeriod;
   }
   
   if (avg_loss != 0)
      rs = avg_gain / avg_loss;
   else
      rs = 0;
   
   ExtRSIBuffer[rates_total-1] = 100.0 - (100.0 / (1.0 + rs));
   
   return(rates_total);
}
```

## MACD

### Basic MACD

```mql4
//+------------------------------------------------------------------+
//| MACD Indicator                                                  |
//+------------------------------------------------------------------+
#property copyright "Your Name"
#property link      "https://yoursite.com"
#property version   "1.00"
#property strict

#property indicator_separate_window
#property indicator_minimum -5
#property indicator_maximum 5

input int FastEMA = 12;
input int SlowEMA = 26;
input int SignalPeriod = 9;

double ExtMacdBuffer[];
double ExtSignalBuffer[];
double ExtHistBuffer[];

int OnInit()
{
   SetIndexBuffer(0, ExtMacdBuffer, INDICATOR_CALCULATIONS);
   SetIndexBuffer(1, ExtSignalBuffer, INDICATOR_CALCULATIONS);
   SetIndexBuffer(2, ExtHistBuffer, INDICATOR_CALCULATIONS);
   
   SetIndexStyle(0, DRAW_LINE);
   SetIndexStyle(1, DRAW_LINE);
   SetIndexStyle(2, DRAW_HISTOGRAM);
   
   SetIndexLabel(0, "MACD");
   SetIndexLabel(1, "Signal");
   SetIndexLabel(2, "Histogram");
   
   ArraySetAsSeries(ExtMacdBuffer, true);
   ArraySetAsSeries(ExtSignalBuffer, true);
   ArraySetAsSeries(ExtHistBuffer, true);
   
   return(INIT_SUCCEEDED);
}

int OnCalculate(const int rates_total,
                const int prev_calculated,
                const datetime &time[],
                const double &open[],
                const high[],
                const low[],
                const double &close[],
                const long &volume[],
                const int &spread[])
{
   if (rates_total < SlowEMA)
      return(0);
   
   double emaFast, emaSlow, macd;
   
   for (int i = 0; i < rates_total; i++)
   {
      emaFast = iCustom(close, i, "MACD Fast EMA", FastEMA, 0);
      emaSlow = iCustom(close, i, "MACD Slow EMA", SlowEMA, 0);
      macd = emaFast - emaSlow;
      
      ExtMacdBuffer[i] = macd;
      
      if (i == 0)
         ExtSignalBuffer[i] = macd;
      else
         ExtSignalBuffer[i] = ExtSignalBuffer[i-1] + 
                               2.0 / (SignalPeriod + 1) * 
                               (macd - ExtSignalBuffer[i-1]);
      
      ExtHistBuffer[i] = macd - ExtSignalBuffer[i];
   }
   
   return(rates_total);
}
```

### Simple MACD

```mql4
//+------------------------------------------------------------------+
//| Simple MACD                                                     |
//+------------------------------------------------------------------+
#property copyright "Your Name"
#property link      "https://yoursite.com"
#property version   "1.00"
#property strict

#property indicator_separate_window

input int FastEMA = 12;
input int SlowEMA = 26;
input int SignalLine = 9;

double ExtMainBuffer[];
double ExtSignalBuffer[];
double ExtHistBuffer[];

int OnInit()
{
   SetIndexBuffer(0, ExtMainBuffer, INDICATOR_CALCULATIONS);
   SetIndexBuffer(1, ExtSignalBuffer, INDICATOR_CALCULATIONS);
   SetIndexBuffer(2, ExtHistBuffer, INDICATOR_CALCULATIONS);
   
   SetIndexStyle(0, DRAW_LINE);
   SetIndexStyle(1, DRAW_LINE);
   SetIndexStyle(2, DRAW_HISTOGRAM, 2);
   
   IndicatorDigits(Digits);
   
   return(INIT_SUCCEEDED);
}

int OnCalculate(const int rates_total,
                const int prev_calculated,
                const datetime &time[],
                const double &open[],
                const high[],
                const low[],
                const close[],
                const long &volume[],
                const int &spread[])
{
   int i, limit;
   
   if (prev_calculated == 0)
   {
      for (i = 0; i < rates_total; i++)
      {
         ExtMainBuffer[i] = 0;
         ExtSignalBuffer[i] = 0;
         ExtHistBuffer[i] = 0;
      }
      limit = rates_total - SlowEMA - 2;
   }
   else
      limit = rates_total - prev_calculated - 1;
   
   double EMAF, EMAS, MA;
   
   for (i = limit; i < rates_total; i++)
   {
      EMAF = iMA(NULL, 0, FastEMA, 0, MODE_EMA, PRICE_CLOSE, i);
      EMAS = iMA(NULL, 0, SlowEMA, 0, MODE_EMA, PRICE_CLOSE, i);
      
      ExtMainBuffer[i] = EMAF - EMAS;
      
      if (i == 0)
         ExtSignalBuffer[i] = ExtMainBuffer[i];
      else
         ExtSignalBuffer[i] = ExtSignalBuffer[i-1] + 
                             2.0 / (SignalLine + 1) * 
                             (ExtMainBuffer[i] - ExtSignalBuffer[i-1]);
      
      ExtHistBuffer[i] = ExtMainBuffer[i] - ExtSignalBuffer[i];
   }
   
   return(rates_total);
}
```

## Stochastic Oscillator

### Basic Stochastic

```mql4
//+------------------------------------------------------------------+
//| Stochastic Oscillator                                            |
//+------------------------------------------------------------------+
#property copyright "Your Name"
#property link      "https://yoursite.com"
#property version   "1.00"
#property strict

#property indicator_separate_window
#property indicator_minimum 0
#property indicator_maximum 100

input int KPeriod = 14;
input int DPeriod = 3;
input int Slowing = 3;

double ExtKBuffer[];
double ExtDBuffer[];

int OnInit()
{
   SetIndexBuffer(0, ExtKBuffer, INDICATOR_CALCULATIONS);
   SetIndexBuffer(1, ExtDBuffer, INDICATOR_CALCULATIONS);
   
   SetIndexStyle(0, DRAW_LINE);
   SetIndexStyle(1, DRAW_LINE);
   
   SetIndexLabel(0, "%K");
   SetIndexLabel(1, "%D");
   
   IndicatorDigits(2);
   
   return(INIT_SUCCEEDED);
}

int OnCalculate(const int rates_total,
                const int prev_calculated,
                const datetime &time[],
                const double &open[],
                const high[],
                const low[],
                const close[],
                const long &volume[],
                const int &spread[])
{
   int i, limit;
   
   if (rates_total < KPeriod)
      return(0);
   
   if (prev_calculated == 0)
   {
      for (i = 0; i < rates_total; i++)
      {
         ExtKBuffer[i] = 0;
         ExtDBuffer[i] = 0;
      }
      limit = KPeriod + Slowing;
   }
   else
      limit = prev_calculated - 1;
   
   double lowes, highes, close1, k;
   
   for (i = limit; i < rates_total; i++)
   {
      lowes = low[i];
      for (int j = 0; j < KPeriod; j++)
      {
         if (low[i-j] < lowes)
            lowes = low[i-j];
      }
      
      highes = high[i];
      for (int j = 0; j < KPeriod; j++)
      {
         if (high[i-j] > highes)
            highes = high[i-j];
      }
      
      k = (highes - lowes != 0) ? 
          ((close[i] - lowes) / (highes - lowes)) * 100 : 50;
      
      ExtKBuffer[i] = k;
      ExtDBuffer[i] = iMA(NULL, 0, DPeriod, 0, MODE_SMA, ExtKBuffer, i);
   }
   
   return(rates_total);
}
```

## CCI (Commodity Channel Index)

```mql4
//+------------------------------------------------------------------+
//| CCI Indicator                                                   |
//+------------------------------------------------------------------+
#property copyright "Your Name"
#property link      "https://yoursite.com"
#property version   "1.00"
#property strict

#property indicator_separate_window

input int CCIPeriod = 20;

double ExtCCIBuffer[];
double ExtTPBuffer[];

int OnInit()
{
   SetIndexBuffer(0, ExtCCIBuffer, INDICATOR_CALCULATIONS);
   SetIndexStyle(0, DRAW_LINE);
   SetIndexLabel(0, "CCI");
   IndicatorDigits(2);
   
   ArraySetAsSeries(ExtCCIBuffer, true);
   ArraySetAsSeries(ExtTPBuffer, true);
   
   return(INIT_SUCCEEDED);
}

int OnCalculate(const int rates_total,
                const int prev_calculated,
                const datetime &time[],
                const double &open[],
                const high[],
                const low[],
                const close[],
                const long &volume[],
                const int &spread[])
{
   if (rates_total < CCIPeriod)
      return(0);
   
   int i, limit;
   double tp, sma, sum;
   
   if (prev_calculated == 0)
   {
      for (i = 0; i < CCIPeriod; i++)
      {
         ExtCCIBuffer[i] = 0;
         ExtTPBuffer[i] = (high[i] + low[i] + close[i]) / 3.0;
      }
      limit = CCIPeriod;
   }
   else
      limit = prev_calculated - 1;
   
   for (i = limit; i < rates_total; i++)
   {
      ExtTPBuffer[i] = (high[i] + low[i] + close[i]) / 3.0;
      
      sum = 0;
      for (int j = 0; j < CCIPeriod; j++)
         sum += MathAbs(ExtTPBuffer[i-j] - ExtTPBuffer[i]);
      
      sma = 0;
      for (int j = 0; j < CCIPeriod; j++)
         sma += ExtTPBuffer[i-j];
      sma /= CCIPeriod;
      
      double mean_dev = sum / CCIPeriod;
      
      if (mean_dev != 0)
         ExtCCIBuffer[i] = (ExtTPBuffer[i] - sma) / (0.015 * mean_dev);
   }
   
   return(rates_total);
}
```

## Williams %R

```mql4
//+------------------------------------------------------------------+
//| Williams %R                                                     |
//+------------------------------------------------------------------+
#property copyright "Your Name"
#property link      "https://yoursite.com"
#property version   "1.00"
#property strict

#property indicator_separate_window
#property indicator_minimum -100
#property indicator_maximum 0

input int WPeriod = 14;

double ExtWRBuffer[];

int OnInit()
{
   SetIndexBuffer(0, ExtWRBuffer, INDICATOR_CALCULATIONS);
   SetIndexStyle(0, DRAW_LINE);
   SetIndexLabel(0, "%R");
   IndicatorDigits(2);
   
   return(INIT_SUCCEEDED);
}

int OnCalculate(const int rates_total,
                const int prev_calculated,
                const datetime &time[],
                const double &open[],
                const high[],
                const low[],
                const close[],
                const long &volume[],
                const int &spread[])
{
   if (rates_total < WPeriod)
      return(0);
   
   int limit = (prev_calculated > WPeriod) ? prev_calculated : WPeriod;
   
   double highest, lowest;
   
   for (int i = limit - 1; i < rates_total; i++)
   {
      highest = high[i];
      lowest = low[i];
      
      for (int j = 0; j < WPeriod; j++)
      {
         if (high[i-j] > highest)
            highest = high[i-j];
         if (low[i-j] < lowest)
            lowest = low[i-j];
      }
      
      if (highest != lowest)
         ExtWRBuffer[i] = -100.0 * ((highest - close[i]) / (highest - lowest));
      else
         ExtWRBuffer[i] = -50.0;
   }
   
   return(rates_total);
}
```

## Complete All-In-One Indicator

```mql4
//+------------------------------------------------------------------+
//| Complete Oscillator Suite                                       |
//+------------------------------------------------------------------+
#property copyright "Your Name"
#property link      "https://yoursite.com"
#property version   "1.00"
#property strict

#property indicator_separate_window
#property indicator_minimum 0
#property indicator_maximum 100

// RSI Inputs
input int RSIPeriod = 14;
input double RSIOverbought = 70.0;
input double RSIOversold = 30.0;

// MACD Inputs
input int FastEMA = 12;
input int SlowEMA = 26;
input int SignalPeriod = 9;

// Stochastic Inputs
input int KPeriod = 14;
input int DPeriod = 3;

// CCI Inputs
input int CCIPeriod = 20;

// Buffers
double RSIBuffer[];
double MACDBuffer[];
double SignalBuffer[];
double HistBuffer[];
double KBuffer[];
double DBuffer[];
double CCIBuffer[];

int OnInit()
{
   // RSI
   SetIndexBuffer(0, RSIBuffer, INDICATOR_CALCULATIONS);
   SetIndexStyle(0, DRAW_LINE);
   SetIndexLabel(0, "RSI");
   
   // MACD
   SetIndexBuffer(1, MACDBuffer, INDICATOR_CALCULATIONS);
   SetIndexBuffer(2, SignalBuffer, INDICATOR_CALCULATIONS);
   SetIndexBuffer(3, HistBuffer, INDICATOR_CALCULATIONS);
   SetIndexStyle(1, DRAW_LINE);
   SetIndexStyle(2, DRAW_LINE);
   SetIndexStyle(3, DRAW_HISTOGRAM);
   
   // Stochastic
   SetIndexBuffer(4, KBuffer, INDICATOR_CALCULATIONS);
   SetIndexBuffer(5, DBuffer, INDICATOR_CALCULATIONS);
   SetIndexStyle(4, DRAW_LINE);
   SetIndexStyle(5, DRAW_LINE);
   
   // CCI
   SetIndexBuffer(6, CCIBuffer, INDICATOR_CALCULATIONS);
   SetIndexStyle(6, DRAW_LINE);
   
   IndicatorDigits(5);
   
   return(INIT_SUCCEEDED);
}

int OnCalculate(const int rates_total,
               const int prev_calculated,
               const datetime &time[],
               const double &open[],
               const high[],
               const low[],
               const double &close[],
               const long &volume[],
               const int &spread[])
{
   // Calculate RSI
   double avg_gain = 0, avg_loss = 0, rs;
   
   for (int i = MathMax(1, rates_total - RSIPeriod); i < rates_total; i++)
   {
      double change = close[i] - close[i-1];
      if (change > 0) avg_gain += change;
      else avg_loss += (-change);
   }
   
   avg_gain /= RSIPeriod;
   avg_loss /= RSIPeriod;
   
   if (avg_loss != 0)
      rs = avg_gain / avg_loss;
   else
      rs = 1;
   
   RSIBuffer[rates_total-1] = 100.0 - (100.0 / (1.0 + rs));
   
   // MACD
   double emaFast = iMA(NULL, 0, FastEMA, 0, MODE_EMA, PRICE_CLOSE, rates_total-1);
   double emaSlow = iMA(NULL, 0, SlowEMA, 0, MODE_EMA, PRICE_CLOSE, rates_total-1);
   
   MACDBuffer[rates_total-1] = emaFast - emaSlow;
   
   if (rates_total > 1)
      SignalBuffer[rates_total-1] = SignalBuffer[rates_total-2] + 
                                   2.0 / (SignalPeriod + 1) * 
                                   (MACDBuffer[rates_total-1] - SignalBuffer[rates_total-2]);
   else
      SignalBuffer[rates_total-1] = MACDBuffer[rates_total-1];
   
   HistBuffer[rates_total-1] = MACDBuffer[rates_total-1] - SignalBuffer[rates_total-1];
   
   // Stochastic
   double highest = high[rates_total-1];
   double lowest = low[rates_total-1];
   
   for (int j = 0; j < KPeriod; j++)
   {
      if (high[rates_total-1-j] > highest) highest = high[rates_total-1-j];
      if (low[rates_total-1-j] < lowest) lowest = low[rates_total-1-j];
   }
   
   if (highest != lowest)
      KBuffer[rates_total-1] = 100.0 * ((close[rates_total-1] - lowest) / (highest - lowest));
   else
      KBuffer[rates_total-1] = 50.0;
   
   // CCI
   double tp = (high[rates_total-1] + low[rates_total-1] + close[rates_total-1]) / 3.0;
   double sma = 0;
   for (int i = 0; i < CCIPeriod; i++)
      sma += (high[rates_total-1-i] + low[rates_total-1-i] + close[rates_total-1-i]) / 3.0;
   sma /= CCIPeriod;
   
   double sum_dev = 0;
   for (int i = 0; i < CCIPeriod; i++)
      sum_dev += MathAbs(tp - ((high[rates_total-1-i] + low[rates_total-1-i] + close[rates_total-1-i]) / 3.0));
   
   double mean_dev = sum_dev / CCIPeriod;
   CCIBuffer[rates_total-1] = (mean_dev != 0) ? (tp - sma) / (0.015 * mean_dev) : 0;
   
   return(rates_total);
}
```

## EA Using Oscillators

### RSI EA Example

```mql4
//+------------------------------------------------------------------+
//| RSI Trading EA                                                  |
//+------------------------------------------------------------------+
#property copyright "Your Name"
#property link      "https://yoursite.com"
#property version   "1.00"
#property strict

input double Lots = 0.1;
input int RSIPeriod = 14;
input double Overbought = 70.0;
input double Oversold = 30.0;
input int Slippage = 3;
input int MagicNumber = 12345;

double RSI;

int OnInit()
{
   return(INIT_SUCCEEDED);
}

void OnDeinit(const int reason)
{
}

void OnTick()
{
   if (NewBar() == false)
      return;
   
   RSI = iRSI(NULL, 0, RSIPeriod, PRICE_CLOSE, 0);
   double prevRSI = iRSI(NULL, 0, RSIPeriod, PRICE_CLOSE, 1);
   
   bool longCondition = (prevRSI <= Oversold) && (RSI > Oversold);
   bool shortCondition = (prevRSI >= Overbought) && (RSI < Overbought);
   
   if (longCondition)
      OpenPosition(OP_BUY, Lots);
   else if (shortCondition)
      OpenPosition(OP_SELL, Lots);
   
   ManagePositions();
}

bool NewBar()
{
   static datetime prevBar = 0;
   datetime currBar = Time[0];
   
   if (prevBar == 0)
   {
      prevBar = currBar;
      return(true);
   }
   
   if (currBar > prevBar)
   {
      prevBar = currBar;
      return(true);
   }
   
   return(false);
}

void OpenPosition(int type, double lot)
{
   string symbol = Symbol();
   double price = (type == OP_BUY) ? Ask : Bid;
   int digits = (int)MarketInfo(symbol, MODE_DIGITS);
   
   string comment = "RSI EA " + IntegerToString(MagicNumber);
   
   int ticket = OrderSend(symbol, type, lot, price, Slippage, 
                         0, 0, comment, MagicNumber, 0, 
                         (type == OP_BUY) ? clrGreen : clrRed);
   
   if (ticket > 0)
   {
      if (OrderSelect(ticket, SELECT_BY_TICKET))
      {
         Print("Order opened: ", OrderTicket(), " Type: ", 
              (type == OP_BUY) ? "BUY" : "SELL");
      }
   }
   else
   {
      Print("Order failed with error: ", GetLastError());
   }
}

void ManagePositions()
{
   for (int i = 0; i < OrdersTotal(); i++)
   {
      if (OrderSelect(i, SELECT_BY_POS))
      {
         if (OrderMagicNumber() == MagicNumber && 
             OrderSymbol() == Symbol())
         {
            double profit = OrderProfit();
            
            if (profit > 0)
            {
               bool close = OrderClose(OrderTicket(), 
                                    OrderLots(), 
                                    (OrderType() == OP_BUY) ? Bid : Ask, 
                                    Slippage, clrNONE);
            }
         }
      }
   }
}
```

## Conclusion

MQL4 provides the following functions for oscillators:
- RSI: `iRSI()`
- MACD: `iMACD()`
- Stochastic: `iStochastic()`
- CCI: `iCCI()`
- Williams %R: Not built-in, must be calculated manually

Key points:
- Buffer management handled with `SetIndexBuffer()`
